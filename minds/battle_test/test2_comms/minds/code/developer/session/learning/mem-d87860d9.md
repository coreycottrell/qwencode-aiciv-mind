```python
from typing import Dict

def fibonacci(n: int, memo: Dict[int, int] = None) -> int:
    """
    Compute the nth Fibonacci number using recursion with memoization.

    Args:
        n: The index of the Fibonacci number to compute (must be positive).
        memo: A dictionary to store computed Fibonacci numbers for memoization.
              If None, a new dictionary is created.

    Returns:
        The nth Fibonacci number.

    Raises:
        ValueError: If n is not a positive integer.
    """
    if memo is None:
        memo = {}
    if n in memo:
        return memo[n]
    if n <= 0:
        raise ValueError("n must be a positive integer")
    if n <= 2:
        return 1
    memo[n] = fibonacci(n - 1, memo) + fibonacci(n - 2, memo)
    return memo[n]
```
